```typescript
import { useState, useEffect } from 'react';

export function useDebounce<T>(value: T, delay = 500): T {
  const [debouncedValue, setDebouncedValue] = useState(value);
  useEffect(() => {
    const handler = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(handler);
  }, [value, delay]);
  return debouncedValue;
}
```

Pattern: Named export, NO 'use client' directive (useDebounce doesn't have it),
generic typed, no external deps, pure React hooks.

NOTE: The research recommended 'use client' but the existing hook (useDebounce.ts)
does NOT have it. Follow the existing codebase convention — omit 'use client' in hooks
unless the build requires it.
